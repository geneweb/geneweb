type date =
  1. | Dgreg of dmy * calendar
  2. | Dtext of string

Date data type that can be either concrete date associated to a calendar or a textual form of the date.

and calendar =
  1. | Dgregorian
  2. | Djulian
  3. | Dfrench
  4. | Dhebrew

Supported calendars

and dmy = {
  1. day : int;
  2. month : int;
  3. year : int;
  4. prec : precision;
  5. delta : int;
}

Concrete date with precision.

and dmy2 = {
  1. day2 : int;
  2. month2 : int;
  3. year2 : int;
  4. delta2 : int;
}

Concrete date without precision.

and precision =
  1. | Sure
  2. | About
  3. | Maybe
  4. | Before
  5. | After
  6. | OrYear of dmy2
  7. | YearInt of dmy2

Precision attached to the concrete date.
