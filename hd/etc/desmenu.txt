function no_options() {
  document.querySelectorAll(
    'input[type="checkbox"]:not([name="only"]):not([name="image"]):not([name="cgl"])'
  ).forEach(function(cb) { cb.checked = false; });
}

function long_display_options() {
  const ldAll      = document.querySelectorAll('input[type="checkbox"][id^="ld_"]');
  const ldUnchecked = document.querySelectorAll('input[type="checkbox"][id^="ld_"]:not(:checked)');
  if (ldAll.length === ldUnchecked.length) {
    %if;(b.always_surname="yes")
    const alwsurn = document.getElementById('ld_alwsurn');
    if (alwsurn) alwsurn.checked = true;
    %end;
    const notes = document.getElementById('ld_notes');
    const src   = document.getElementById('ld_src');
    if (notes) notes.checked = true;
    if (src)   src.checked   = true;
  }
}

function table_options() {
  /* Appelé après no_options() — on coche directement les defaults sans guard. */
  const tmarr = document.getElementById('t_marr');
  if (tmarr) { tmarr.disabled = false; tmarr.checked = true; }
  ['t_num_table','t_birth_date','t_birth_place',
   't_marr_place','t_marr_date','t_child',
   't_death_date','t_death_place','t_occu','t_gen'
  ].forEach(function(id) {
    const el = document.getElementById(id);
    if (el) el.checked = true;
  });
}

function table_spouse_disabled() {
  /* Appelé après table_options() pour le mode "tableau avec infos conjoints". */
  const tmarr  = document.getElementById('t_marr');
  const tmarr2 = document.getElementById('t_marr2');
  if (tmarr)  { tmarr.disabled = true;  tmarr.checked = true; }
  if (tmarr2) { tmarr2.disabled = false; }
}
