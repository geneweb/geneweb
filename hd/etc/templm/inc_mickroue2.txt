          %apply;display_images_marriage("xx", "yy", "zz")

        %apply;display_images_person("xx", "yy", "zz")
      %end;
      %apply;display_images%with;
        xx%and;
        %apply;letter(curlev + 2)%and;
        zz%and;
        %expr(curlev + 1)%and;
        %maxlev;
      %end;
    %end;
  %end;
