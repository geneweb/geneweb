<script>
// ── Helpers ────────────────────────────────────────────────────────────────

const $ = id => document.getElementById(id);
const $q = sel => document.querySelector(sel);
const $qa = sel => document.querySelectorAll(sel);

function setDisabled(els, state) {
  els.forEach(el => { if (el) el.disabled = state; });
}

function formatButtonText(noteText, imageName, maxLen = 37) {
  const ext = imageName.lastIndexOf('.');
  const name = ext !== -1 ? imageName.slice(0, ext) : imageName;
  const truncName = name.length > maxLen ? name.slice(0, maxLen - 3) + '[…]' : name;
  const suffix = ext !== -1 ? imageName.slice(ext) : '';
  const whichEl = $('which_img_show');
  if (whichEl) whichEl.style.whiteSpace = name.length > maxLen ? 'pre-line' : 'normal';
  return `${noteText}\n${truncName}${suffix}`;
}

function updateTextareaSize() {
  // autosize est optionnel — présent sur certaines pages seulement
  if (typeof autosize !== 'undefined') {
    setTimeout(() => autosize.update($qa('textarea')), 0);
  }
}

let editAbort = null;

// ── File input BS5 (remplace custom-file BS4) ──────────────────────────────
// Pour portrait et blason : active simplement le bouton submit sur changement de fichier.
// Pour "others" (carrousel) : active le btn-quick-upload + les champs note/source.
function setupFileInput(fileInputSel, submitSel, isCarrousel) {
  const input = $q(fileInputSel);
  const btn   = $q(submitSel);
  if (!input || !btn) return;

  input.addEventListener('change', () => {
    const hasFile = input.value !== '';
    btn.disabled = !hasFile;
    if (isCarrousel && hasFile) {
      setDisabled([$('image_note'), $('image_source')], false);
      setDisabled([$q('.submit-button')], false);
      // URL et nom désactivés quand un fichier est sélectionné
      const urlEl  = $('image_url');
      const nameEl = $('image_name');
      if (urlEl)  { urlEl.disabled = true;  urlEl.value = ''; }
      if (nameEl) { nameEl.disabled = true; nameEl.value = ''; }
      showCancelBtn();
      const note = $('image_note');
      if (note) note.focus();
    }
  });
}

function showCancelBtn() {
  const btn = $('btn-cancel-edit');
  if (btn) { btn.classList.remove('d-none'); btn.classList.add('d-flex', 'align-items-center'); }
}

function hideCancelBtn() {
  const btn = $('btn-cancel-edit');
  if (btn) { btn.classList.add('d-none'); btn.classList.remove('d-flex', 'align-items-center'); }
}

// ── Édition de note/source sur image existante ────────────────────────────
const editImageNote = (cnt, name) => {
  const anchor    = $(`source_${cnt}`);
  const noteData  = anchor ? anchor.dataset.note   : '';
  const srcData   = anchor ? anchor.dataset.source : '';

  // Verrouiller upload pendant l'édition
  const othersFile = $('others_file');
  const urlEl      = $('image_url');
  const nameEl     = $('image_name');
  if (othersFile) othersFile.disabled = true;
  if (urlEl)      urlEl.disabled = true;
  if (nameEl)     nameEl.disabled = true;
  $q('.btn-quick-upload').disabled = true;

  showCancelBtn();

  const noteEl = $('image_note');
  const srcEl  = $('image_source');
  if (noteEl) { noteEl.value = noteData || ''; noteEl.disabled = false; }
  if (srcEl)  { srcEl.value  = srcData  || ''; srcEl.disabled  = false; }
  updateTextareaSize();
  if (noteEl) noteEl.focus();

  $('file_name_2').value = name;
  $('mode_2').value      = 'note';

  const whichEl = $('which_img_show');
  if (whichEl) {
    whichEl.textContent = formatButtonText(
      noteData ? '[*update image note]' : '[*add note]/[source/sources]0',
      name
    );
  }
  const submitBtn = $q('.submit-button');
  if (submitBtn) submitBtn.disabled = false;

  // Nettoyer les listeners précédents et ré-évaluer le submit selon saisie
  if (editAbort) editAbort.abort();
  editAbort = new AbortController();
  [noteEl, srcEl].forEach(el => {
    if (!el) return;
    el.addEventListener('input', () => {
      if (submitBtn) submitBtn.disabled = !el.value.length;
      updateTextareaSize();
    }, { signal: editAbort.signal });
  });

  return false;
};

// ── URL d'image externe ────────────────────────────────────────────────────
$('image_url')?.addEventListener('input', function() {
  const url    = this.value.trim();
  const nameEl = $('image_name');
  const submitBtn  = $q('.submit-button');
  const quickBtn   = $q('.btn-quick-upload');
  const othersFile = $('others_file');

  if (url) {
    if (quickBtn)   quickBtn.disabled   = false;
    if (submitBtn)  submitBtn.disabled  = false;
    setDisabled([$('image_note'), $('image_source')], false);
    if (othersFile) { othersFile.disabled = true; }
    if (nameEl)     { nameEl.disabled = false; }
    // Pré-remplir le placeholder du nom depuis l'URL
    if (nameEl && !nameEl.value) {
      const filename = url.split('/').pop().split('?')[0];
      if (filename && filename.includes('.')) nameEl.placeholder = filename + ' (auto)';
    }
    showCancelBtn();
  } else {
    if (othersFile && !othersFile.disabled) {
      if (quickBtn)  quickBtn.disabled  = true;
      if (submitBtn) submitBtn.disabled = true;
      setDisabled([$('image_note'), $('image_source')], true);
      hideCancelBtn();
    }
    if (nameEl) { nameEl.disabled = true; nameEl.value = ''; }
  }
});

// ── Champs métadonnées (nom/note/source) ──────────────────────────────────
['image_name', 'image_note', 'image_source'].forEach(id => {
  $(id)?.addEventListener('input', function() {
    const othersFile = $('others_file');
    const urlEl      = $('image_url');
    const submitBtn  = $q('.submit-button');
    if (submitBtn && (
      (othersFile && othersFile.value) ||
      (urlEl && urlEl.value.trim()) ||
      this.value.trim()
    )) submitBtn.disabled = false;
  });
});

// ── Annulation d'édition ──────────────────────────────────────────────────
$('btn-cancel-edit')?.addEventListener('click', () => {
  setDisabled([$('image_note'), $('image_source')], true);
  const noteEl = $('image_note');
  const srcEl  = $('image_source');
  if (noteEl) noteEl.value = '';
  if (srcEl)  srcEl.value  = '';

  $('file_name_2').value = '';
  $('mode_2').value      = 'carrousel';

  const whichEl = $('which_img_show');
  if (whichEl) whichEl.textContent = '[*send image with note and source]';

  const submitBtn = $q('.submit-button');
  if (submitBtn) submitBtn.disabled = true;

  const othersFile = $('others_file');
  if (othersFile) { othersFile.value = ''; othersFile.disabled = false; }

  const urlEl  = $('image_url');
  const nameEl = $('image_name');
  if (urlEl)  { urlEl.value  = ''; urlEl.disabled  = false; }
  if (nameEl) { nameEl.value = ''; nameEl.disabled = true; }

  const quickBtn = $q('.btn-quick-upload');
  if (quickBtn) quickBtn.disabled = true;

  hideCancelBtn();
});

// ── Btn-quick-upload ──────────────────────────────────────────────────────
$q('.btn-quick-upload')?.addEventListener('click', () => {
  const othersFile = $('others_file');
  if (othersFile && othersFile.value) othersFile.closest('form').submit();
});

// ── Submit guard ──────────────────────────────────────────────────────────
$qa('form').forEach(form => {
  form.addEventListener('submit', () => {
    if ($('mode_2')?.value !== 'note') {
      setTimeout(() => setDisabled([$('image_note'), $('image_source')], true), 100);
    }
  });
});

// ── Tooltips BS5 ──────────────────────────────────────────────────────────
document.addEventListener('DOMContentLoaded', () => {
  setupFileInput('.custom-portrait', '.snd-btn-portrait', false);
  setupFileInput('.custom-blason',   '.snd-btn-blason',   false);
  setupFileInput('.custom-others',   '.btn-quick-upload', true);

  $qa('[data-bs-toggle="tooltip"]').forEach(el => {
    new bootstrap.Tooltip(el, { html: true });
  });
});
</script>
