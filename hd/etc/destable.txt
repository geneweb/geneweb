%define;one_child(xx, prefx)

%define;one_family(xx, prefx)

%define;one_level(xx, lv, max_l, prefx, ch_cnt)

        %end;
      %end;
    %else;
    %end
  %else;
    %apply;one_family("self", npref)
  %end;
%end;
