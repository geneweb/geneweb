'use strict';

const MAPHILIGHT_CONFIG = {
  fillColor: 'ff0000',
  fillOpacity: 0.15,
  strokeColor: 'ff0000',
  strokeOpacity: 0.6,
  strokeWidth: 2,
  alwaysOn: false
};

// palette alternant chaud/froid/pastel + bon contraste avec bleu #007bbf (0.3 opacity)
const GROUP_COLORS = %[
  'ff6b6b', 'a0c4ff', 'ffd93d', 'c7ceea', 'ffaaa5', '4ecdc4',
  'fdcb6e', 'ff8b94', '74b9ff', 'ffc3a0', 'a8e6cf', 'ffc6ff',
  'b4f8c8', 'ff7675', '89e5e5', 'ffeaa7', 'd4a5ff', 'fab1a0',
  'c7f0db', 'fd79a8', 'fff3cd', 'ffccf9', 'ffe0f0', 'ffd9e8'
%];

let current = 0;
let imagesCount = 0;
let json, images_l;
const groupLocked = {};      // { group1: true, group2: false, ... }
const lockedColors = {};     // { area3: 'ff6b6b', ... } — par id d'area
let colorOffset = 0;

// --- Small DOM helpers -------------------------------------------------
const $$  = (sel, root) => (root || document).querySelector(sel);
const $$$ = (sel, root) => (root || document).querySelectorAll(sel);
const get = v => v !== undefined ? v : '';

function emptyNode(el) { while (el.firstChild) el.removeChild(el.firstChild); }

// Highlight config pour une area donnée, en tenant compte de son état locké.
// Retourne la config à passer à maphilight.set().
function configFor(area, opts) {
  opts = opts || {};
  const color = lockedColors[area.id];
  if (color) {
    return {
      ...MAPHILIGHT_CONFIG,
      alwaysOn: true,
      fillColor: color,
      strokeColor: color,
      fillOpacity: opts.boost ? 0.5 : 0.2,
      groupBy: false
    };
  }
  if (opts.hover) {
    return { ...MAPHILIGHT_CONFIG, alwaysOn: true, groupBy: false };
  }
  return MAPHILIGHT_CONFIG;
}

// Met à jour la couleur de fond du label de légende associé à une area.
function paintLegend(areaId, color) {
  const legendLink = $$('#' + areaId.replace('area', 'a') + ' span a');
  if (legendLink) legendLink.style.backgroundColor = color || '';
}

// Unifie l'entrée en highlight (hover area OU hover legend item)
function enterArea(area) {
  const color = lockedColors[area.id];
  maphilight.set(area, configFor(area, { hover: true, boost: !!color }));
  paintLegend(area.id, color ? '#' + color + 'aa' : '#ffccaa');
}

function leaveArea(area) {
  const color = lockedColors[area.id];
  maphilight.set(area, configFor(area));
  paintLegend(area.id, color ? '#' + color + '66' : '');
}

// --- Image rendering ---------------------------------------------------

function displayImage(img) {
  const map = $$('#imap-notes');
  const legend = $$('#legend');
  emptyNode(map);
  emptyNode(legend);
  $$('#div_legend').classList.add('d-none');
  $$('#h1').innerHTML = json.title;
  $$('#description').innerHTML = img.desc || json.desc || '';
  const imageEl = $$('#image');
  imageEl.src = '%prefix;m=DOC&s=' + img.img;
  imageEl.hidden = false;

  let cnt = 1, cnt_rlm = 1;
  const hasGroups = img.groups?.length;
  const ungrouped = [];
  const rlmEl = $$('#rlm');

  img.map.forEach(r => {
    const t = get(r.t) || 'p';
    let href, txt, rlmParams = '';
    let comment = get(r.alt) ? ` (${r.alt})` : '';
    if (t === 'p') {
      if (!r.fn && !r.sn) { cnt++; return; }
      const oc = get(r.oc) && r.oc != '0' ? `&oc=${r.oc}` : '';
      href = `%prefix;&p=${r.fn}&n=${r.sn}${oc}`;
      txt = `${r.fn} ${r.sn}`;
      rlmParams = `&p${cnt_rlm}=${r.fn}&n${cnt_rlm}=${r.sn}${oc}`;
      cnt_rlm++;
    } else if (t === 'g') {
      href = `%prefix;${get(r.href)}`;
      txt = get(r.alt) || get(r.href);
      comment = '';
    } else if (t === 'e') {
      href = get(r.href);
      txt = get(r.alt) || get(r.href);
      comment = '';
    } else { cnt++; return; }

    const groupNum = r.group || 0;
    const groupClass = (hasGroups && groupNum) ? ` class='group${groupNum}'` : '';
    const areaHtml =
      `<area id='area${cnt}' shape='${get(r.shape)}' coords='${get(r.coords)}'` +
      ` alt='${txt}${comment}' title='${txt}${comment}'` +
      ` href='${href}'${groupClass} target='_blank'>`;
    const legendHtml =
      `<li class='legend' id='a${cnt}'>` +
      `<span><a href="${href}" target='_blank'>${txt}</a>${comment}</span></li>`;

    if (rlmParams) {
      rlmEl.setAttribute('href', rlmEl.getAttribute('href') + rlmParams);
    }

    const item = { id: cnt, areaHtml, legendHtml, groupNum };

    if (hasGroups && groupNum) {
      const bucket = img.groups.find(g => g.name == groupNum);
      if (bucket) {
        (bucket._items ||= []).push(item);
      } else {
        ungrouped.push(item);
      }
    } else {
      ungrouped.push(item);
    }
    cnt++;
  });

  // Render grouped items
  if (hasGroups) {
    img.groups.forEach(g => {
      if (!g._items?.length) return;
      let itemsHtml = '';
      g._items.forEach(item => {
        map.insertAdjacentHTML('beforeend', item.areaHtml);
        itemsHtml += item.legendHtml;
      });
      legend.insertAdjacentHTML('beforeend',
        `<div class="imap-group mb-2">` +
        `<div class="imap-group-legend" id="g${g.name}">` +
        `<a href="#" title="[*lock group]">${g.label}</a></div>` +
        `<ul class="list-comma list-unstyled ps-3">${itemsHtml}</ul></div>`);
      delete g._items;
    });
  }

  // Render ungrouped items
  ungrouped.forEach(item => {
    map.insertAdjacentHTML('beforeend', item.areaHtml);
    legend.insertAdjacentHTML('beforeend', item.legendHtml);
  });

  if (cnt > 1 || ungrouped.length) {
    $$('#div_legend').classList.remove('d-none');
  }
  maphilight(imageEl, MAPHILIGHT_CONFIG);
}

// --- Event delegation --------------------------------------------------

// Hover sur une area (image map)
document.addEventListener('mouseover', e => {
  const area = e.target.closest('area');
  if (area) enterArea(area);
});
document.addEventListener('mouseout', e => {
  const area = e.target.closest('area');
  if (area) leaveArea(area);
});

// Hover sur un item de légende individuel (pas le label de groupe)
document.addEventListener('mouseover', e => {
  const li = e.target.closest('.legend:not(.imap-group-legend)');
  if (!li) return;
  const area = $$('#' + li.id.replace('a', 'area'));
  if (area) enterArea(area);
});
document.addEventListener('mouseout', e => {
  const li = e.target.closest('.legend:not(.imap-group-legend)');
  if (!li) return;
  const area = $$('#' + li.id.replace('a', 'area'));
  if (area) leaveArea(area);
});

// Hover sur un label de groupe : colore toutes ses areas d'une palette tournante
// (effet éphémère : s'annule au mouseout sauf si le groupe est locké)
document.addEventListener('mouseover', e => {
  const label = e.target.closest('.imap-group-legend');
  if (!label) return;
  const groupClass = 'group' + label.id.replace('g', '');
  if (groupLocked[groupClass]) return;

  $$$('.' + groupClass).forEach((area, index) => {
    const color = GROUP_COLORS%[(colorOffset + index) % GROUP_COLORS.length%];
    lockedColors[area.id] = color;
    maphilight.set(area, {
      ...MAPHILIGHT_CONFIG, alwaysOn: true,
      fillColor: color, strokeColor: color, groupBy: false
    });
    paintLegend(area.id, '#' + color + '66');
  });
});

document.addEventListener('mouseout', e => {
  const label = e.target.closest('.imap-group-legend');
  if (!label) return;
  const groupClass = 'group' + label.id.replace('g', '');
  if (groupLocked[groupClass]) return;

  const areas = $$$('.' + groupClass);
  colorOffset += areas.length;
  areas.forEach(area => {
    delete lockedColors[area.id];
    maphilight.set(area, MAPHILIGHT_CONFIG);
    paintLegend(area.id, '');
  });
});

// Clic sur un label de groupe : lock/unlock
document.addEventListener('click', e => {
  const label = e.target.closest('.imap-group-legend');
  if (!label) return;
  e.preventDefault();
  const groupClass = 'group' + label.id.replace('g', '');

  // Si on lock un nouveau groupe, délocker tous les autres d'abord
  if (!groupLocked[groupClass]) {
    Object.keys(groupLocked).forEach(other => {
      if (groupLocked[other]) {
        groupLocked[other] = false;
        $$$('.imap-group-legend').forEach(el => el.classList.remove('imap-locked'));
        $$$('area[class^="group"]').forEach(area => {
          delete lockedColors[area.id];
          maphilight.set(area, MAPHILIGHT_CONFIG);
          paintLegend(area.id, '');
        });
      }
    });
  }

  groupLocked[groupClass] = !groupLocked[groupClass];
  label.classList.toggle('imap-locked');

  const areas = $$$('.' + groupClass);
  if (groupLocked[groupClass]) {
    // Assigner les couleurs si pas déjà fait (cas click direct sans hover préalable)
    const firstArea = areas[0];
    if (firstArea && !lockedColors[firstArea.id]) {
      areas.forEach((area, index) => {
        lockedColors[area.id] = GROUP_COLORS%[(colorOffset + index) % GROUP_COLORS.length%];
      });
      colorOffset += areas.length;
    }
    areas.forEach(area => {
      const color = lockedColors[area.id];
      maphilight.set(area, {
        ...MAPHILIGHT_CONFIG, alwaysOn: true,
        fillColor: color, strokeColor: color,
        fillOpacity: 0.2, groupBy: false
      });
      paintLegend(area.id, '#' + color + '66');
    });
  } else {
    areas.forEach(area => {
      delete lockedColors[area.id];
      maphilight.set(area, MAPHILIGHT_CONFIG);
      paintLegend(area.id, '');
    });
  }
});

// --- Navigation --------------------------------------------------------

function updateNavButtons() {
  const atStart = current <= 0;
  const atEnd   = current >= imagesCount - 1;
  ['#first', '#previous'].forEach(sel => {
    const el = $$(sel);
    if (!el) return;
    el.classList.toggle('disabled', atStart);
    el.classList.toggle('text-body-secondary', atStart);
  });
  ['#next', '#last'].forEach(sel => {
    const el = $$(sel);
    if (!el) return;
    el.classList.toggle('disabled', atEnd);
    el.classList.toggle('text-body-secondary', atEnd);
  });
}

function loadCurrentImage() {
  displayImage(images_l[current]);
  $$('#current').textContent = current + 1;
  $$('#imageCount').textContent = imagesCount;
  $$('#index-wall').classList.remove('d-flex');
  $$('#index-wall').classList.add('d-none');
  $$('#gallery-view').classList.add('d-inline-flex');
  $$('#gallery-view').classList.remove('d-none');
  const nav = $$('#album-nav');
  if (imagesCount > 1) {
    nav.classList.remove('d-none'); nav.classList.add('d-flex');
  } else {
    nav.classList.remove('d-flex'); nav.classList.add('d-none');
  }
  const u = new URL(window.location);
  u.searchParams.set('img', current + 1);
  history.replaceState(null, '', u);
  const edit = $$('#edit_link');
  if (edit) {
    edit.setAttribute('href', '%prefix;m=MOD_NOTES&f=%e.f;&img=' + (current + 1));
  }
  updateNavButtons();
}

function loadIndex() {
  emptyNode($$('#imap-notes'));
  emptyNode($$('#legend'));
  $$('#gallery-view').classList.remove('d-inline-flex');
  $$('#gallery-view').classList.add('d-none');
  const wall = $$('#index-wall');
  emptyNode(wall);
  wall.classList.add('d-flex');
  wall.classList.remove('d-none');
  $$('#h1').innerHTML = json.title;
  const chr = json.chronicle || '';
  const chrEl = $$('#chronicle');
  if (chr) { chrEl.innerHTML = chr; chrEl.classList.remove('d-none'); }
  else { chrEl.classList.add('d-none'); }
  $$('#album-nav').classList.remove('d-flex');
  $$('#album-nav').classList.add('d-none');

  images_l.forEach((img, i) => {
    const desc = img.desc || '';
    const label = desc ? (i + 1) + '. ' + desc : '. ' + (i + 1);
    const url = '%prefix;m=DOC&s=' + img.img;
    const html =
      `<div class="imap-gallery" data-img="${i + 1}"` +
      ` data-bs-toggle="tooltip" data-bs-html="true"` +
      ` title="${label.replace(/"/g, '&quot;')}"` +
      ` style="cursor:pointer">` +
      `<img src="${url}" alt="">` +
      `<div class="gallery-legend">${label}</div></div>`;
    wall.insertAdjacentHTML('beforeend', html);
  });

  // BS5 tooltip init (remplace $.tooltip() de BS4)
  $$$('#index-wall [data-bs-toggle="tooltip"]').forEach(
    el => new bootstrap.Tooltip(el));

  const u = new URL(window.location);
  u.searchParams.delete('img');
  history.replaceState(null, '', u);
  current = -1;
  const edit = $$('#edit_link');
  if (edit) edit.setAttribute('href', '%prefix;m=MOD_NOTES&f=%e.f;');
}

function loadFirstImage()    { if (current <= 0) return; current = 0; loadCurrentImage(); }
function loadLastImage()     { if (current >= imagesCount - 1) return; current = imagesCount - 1; loadCurrentImage(); }
function loadNextImage()     { if (current >= imagesCount - 1) return; current++; loadCurrentImage(); }
function loadPreviousImage() { if (current <= 0) return; current--; loadCurrentImage(); }

// --- Boot --------------------------------------------------------------

document.addEventListener('DOMContentLoaded', () => {
  // Click sur une vignette de l'index
  $$('#index-wall').addEventListener('click', e => {
    const tile = e.target.closest('[data-img]');
    if (!tile) return;
    e.preventDefault();
    current = parseInt(tile.dataset.img, 10) - 1;
    loadCurrentImage();
  });

  fetch('%url;&ajax=on')
    .then(r => {
      if (!r.ok) throw new Error('HTTP ' + r.status);
      return r.json();
    })
    .then(j => {
      json = j;
      images_l = j.images?.length
        ? j.images
        : [{ img: j.img, map: j.map, groups: j.groups }];
      imagesCount = images_l.length;
      document.title = json.title;
      const p = new URLSearchParams(window.location.search);
      const si = parseInt(p.get('img'), 10) || 0;
      if (si > 0 && si <= imagesCount) {
        current = si - 1;
        loadCurrentImage();
      } else if (imagesCount > 1) {
        loadIndex();
      } else {
        current = 0;
        loadCurrentImage();
      }
    })
    .catch(() => { $$('#h1').textContent = '[*loading/error]1'; });
});
