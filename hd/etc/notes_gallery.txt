function get(v) { return v !== undefined ? v : ""; }

function displayImage(img) {
  $("#map, #legend").empty();
  $("#div_legend").addClass("d-none");
  $("#h1").html(json.title);
  $("#description").html(img.desc || json.desc || "");
  $("#image").attr("src", "%prefix;m=DOC&s=" + img.img);
  
  let cnt = 1, cnt_rlm = 1;
  const hasGroups = img.groups?.length;
  const ungrouped = [];

  img.map.forEach(r => {
    const t = get(r.t) || "p";
    let href, txt, rlmParams = "";
    let comment = get(r.alt) ? ` (${r.alt})` : "";
    if (t === "p") {
      if (!r.fn && !r.sn) { cnt++; return; }
      const oc = get(r.oc) && r.oc != "0"
        ? `&oc=${r.oc}` : "";
      href = `%prefix;&p=${r.fn}&n=${r.sn}${oc}`;
      txt = `${r.fn} ${r.sn}`;
      rlmParams =
        `&p${cnt_rlm}=${r.fn}&n${cnt_rlm}=${r.sn}${oc}`;
      cnt_rlm++;
    } else if (t === "g") {
      href = `%prefix;${get(r.href)}`;
      txt = get(r.alt) || get(r.href);
      comment = "";
    } else if (t === "e") {
      href = get(r.href);
      txt = get(r.alt) || get(r.href);
      comment = "";
    } else { cnt++; return; }

    const groupNum = r.group || 0;
    const groupClass = (hasGroups && groupNum)
      ? ` class='group${groupNum}'` : "";
    const areaHtml = `<area id='area${cnt}'`
      + ` shape='${get(r.shape)}'`
      + ` coords='${get(r.coords)}'`
      + ` alt='${txt}${comment}'`
      + ` title='${txt}${comment}'`
      + ` href='${href}'${groupClass}`
      + ` target='_blank'>`;
    const legendHtml = `<li class='legend' id='a${cnt}'>`
      + `<span><a href="${href}" target='_blank'>`
      + `${txt}</a>${comment}</span></li>`;

    if (rlmParams) {
      $("#rlm").attr("href",
        $("#rlm").attr("href") + rlmParams);
    }

    const item = {
      id: cnt, areaHtml: areaHtml,
      legendHtml: legendHtml, groupNum: groupNum
    };

    if (hasGroups && groupNum) {
      const bucket = img.groups.find(
        g => g.name == groupNum);
      if (bucket) {
        if (!bucket._items) bucket._items = [];
        bucket._items.push(item);
      } else {
        ungrouped.push(item);
      }
    } else {
      ungrouped.push(item);
    }
    cnt++;
  });

  // Render grouped items
  if (hasGroups) {
    img.groups.forEach(g => {
      if (!g._items || !g._items.length) return;
      let itemsHtml = "";
      g._items.forEach(item => {
        $("#map").append(item.areaHtml);
        itemsHtml += item.legendHtml;
      });
      $("#legend").append(
        `<div class="imap-group mb-2">`
        + `<div class="imap-group-legend" id="g${g.name}">`
        + `<a href="#" title="[*lock group]">`
        + `${g.label}</a></div>`
        + `<ul class="list-comma list-unstyled pl-3">`
        + `${itemsHtml}</ul></div>`);
      delete g._items;
    });
  }

  // Render ungrouped items
  ungrouped.forEach(item => {
    $("#map").append(item.areaHtml);
    $("#legend").append(item.legendHtml);
  });

  if (cnt > 1 || ungrouped.length) {
    $("#div_legend").removeClass("d-none");
  }
  $("#image").maphilight(MAPHILIGHT_CONFIG);
}

$(document).on("mouseover", "area", function() {
  const $area = $(this);
  const areaId = $area.attr('id');
  const groupClass = $area.attr('class');

  if (groupClass && groupLocked[groupClass]) {
    const lockedColor = $area.data('locked-color');
    if (lockedColor) {
      $area.data('maphilight', {
        ...MAPHILIGHT_CONFIG,
        alwaysOn: true,
        fillColor: lockedColor,
        fillOpacity: 0.5,
        strokeColor: lockedColor,
        groupBy: false
      }).trigger('alwaysOn.maphilight');

      $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '#' + lockedColor + 'aa');
    }
  } else {
    $area.data('maphilight', { ...MAPHILIGHT_CONFIG, alwaysOn: true, groupBy: false }).trigger('alwaysOn.maphilight');
    $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '#ffccaa');
  }
});

let colorOffset = 0;

$(document).on("mouseover", ".imap-group-legend", function() {
  const groupNum = $(this).attr('id').replace('g', '');
  const groupClass = 'group' + groupNum;
  
  if (groupLocked[groupClass]) return;
  
  $('.' + groupClass).each(function(index) {
    const $area = $(this);
    const areaId = $area.attr('id');
    const color = GROUP_COLORS%[(colorOffset + index) % GROUP_COLORS.length%];
    
    $area.data('maphilight', {
      ...MAPHILIGHT_CONFIG,
      alwaysOn: true,
      fillColor: color,
      strokeColor: color,
      groupBy: false
    }).trigger('alwaysOn.maphilight');
    
    $area.data('locked-color', color);
    $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '#' + color + '66');
  });
});

$(document).on("mouseout", ".imap-group-legend", function() {
  const groupNum = $(this).attr('id').replace('g', '');
  const groupClass = 'group' + groupNum;
  
  if (groupLocked[groupClass]) return;
  
  const groupSize = $('.' + groupClass).length;
  colorOffset += groupSize;
  
  $('.' + groupClass).each(function() {
    const $area = $(this);
    const areaId = $area.attr('id');
    
    $area.removeData('locked-color');
    $area.data('maphilight', MAPHILIGHT_CONFIG).trigger('alwaysOn.maphilight');
    $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '');
  });
});

$(document).on("click", ".imap-group-legend", function(e) {
  e.preventDefault();
  const groupNum = $(this).attr('id').replace('g', '');
  const groupClass = 'group' + groupNum;
  
  if (!groupLocked[groupClass]) {
    Object.keys(groupLocked).forEach(otherGroup => {
      if (groupLocked[otherGroup]) {
        groupLocked[otherGroup] = false;
        $('.imap-group-legend').removeClass('imap-locked');
        
        $('area[class^="group"]').each(function() {
          const areaId = $(this).attr('id');
          $(this).removeData('locked-color');
          $(this).data('maphilight', MAPHILIGHT_CONFIG).trigger('alwaysOn.maphilight');
          $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '');
        });
      }
    });
  }
  
  groupLocked[groupClass] = !groupLocked[groupClass];
  $(this).toggleClass('imap-locked');
  
  if (groupLocked[groupClass]) {
    let needsColors = false;
    $('.' + groupClass).first().each(function() {
      if (!$(this).data('locked-color')) needsColors = true;
    });
    
    if (needsColors) {
      $('.' + groupClass).each(function(index) {
        const color = GROUP_COLORS%[(colorOffset + index) % GROUP_COLORS.length%];
        $(this).data('locked-color', color);
      });
      colorOffset += $('.' + groupClass).length;
    }
    
    $('.' + groupClass).each(function() {
      const color = $(this).data('locked-color');
      const areaId = $(this).attr('id');
      
      $(this).data('maphilight', {
        ...MAPHILIGHT_CONFIG,
        alwaysOn: true,
        fillColor: color,
        fillOpacity: 0.2,
        strokeColor: color,
        groupBy: false
      }).trigger('alwaysOn.maphilight');
      
      $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '#' + color + '66');
    });
  } else {
    $('.' + groupClass).each(function() {
      const areaId = $(this).attr('id');
      $(this).removeData('locked-color');
      $(this).data('maphilight', MAPHILIGHT_CONFIG).trigger('alwaysOn.maphilight');
      $('#' + areaId.replace('area', 'a') + ' span a').css('background-color', '');
    });
  }
});

function updateNavButtons() {
  $("#first, #previous").toggleClass("disabled text-muted",
    current <= 0);
  $("#next, #last").toggleClass("disabled text-muted",
    current >= imagesCount - 1);
}

function loadCurrentImage() {
  displayImage(images_l[current]);
  $("#current").text(current + 1);
  $("#imageCount").text(imagesCount);
  $("#index-wall").removeClass("d-flex").addClass("d-none");
  $("#gallery-view").addClass("d-inline-flex").removeClass("d-none");
  if (imagesCount > 1) {
    $("#album-nav").addClass("d-flex");
  }
  var u = new URL(window.location);
  u.searchParams.set("img", current + 1);
  history.replaceState(null, "", u);
  var $edit = $("#edit_link");
  if ($edit.length) {
    var base = "%prefix;m=MOD_NOTES&f=%e.f;";
    $edit.attr("href", base + "&img=" + (current + 1));
  }
  updateNavButtons();
}

function loadIndex() {
  $("#map, #legend").empty();
  $("#gallery-view").removeClass("d-inline-flex").addClass("d-none");
  $("#index-wall").empty().addClass("d-flex").removeClass("d-none");
  $("#h1").html(json.title);
  var chr = json.chronicle || "";
  if (chr) { $("#chronicle").html(chr).removeClass("d-none"); }
  else { $("#chronicle").addClass("d-none"); }
  $("#album-nav").removeClass("d-flex");
  images_l.forEach(function(img, i) {
    var desc = img.desc || "";
    var label = desc ? (i + 1) + ". " + desc : ". " + (i + 1);
    var url = "%prefix;m=DOC&s=" + img.img;
    var html = '<div class="imap-gallery" data-img="' + (i + 1) + '"'
      + ' data-toggle="tooltip" data-html="true"'
      + ' title="' + label.replace(/"/g, '&quot;') + '"'
      + ' style="cursor:pointer">'
      + '<img src="' + url + '" alt="">'
      + '<div class="gallery-legend">' + label + '</div></div>';
    $("#index-wall").append(html);
  });
  $("#index-wall [data-toggle='tooltip']").tooltip();
  var u = new URL(window.location);
  u.searchParams.delete("img");
  history.replaceState(null, "", u);
  current = -1;
  var $edit = $("#edit_link");
  if ($edit.length) {
    $edit.attr("href", "%prefix;m=MOD_NOTES&f=%e.f;");
  }
}

function loadFirstImage() {
  if (current <= 0) return;
  current = 0; loadCurrentImage();
}
function loadLastImage() {
  if (current >= imagesCount - 1) return;
  current = imagesCount - 1; loadCurrentImage();
}
function loadNextImage() {
  if (current >= imagesCount - 1) return;
  current++; loadCurrentImage();
}
function loadPreviousImage() {
  if (current <= 0) return;
  current--; loadCurrentImage();
}

$(document).ready(function() {
  $("#index-wall").on("click", "[data-img]", function(e) {
    e.preventDefault();
    current = parseInt($(this).data("img")) - 1;
    loadCurrentImage();
  });
  $.ajax("%url;&ajax=on")
    .done(function(j) {
      json = j;
      images_l = j.images?.length
        ? j.images
        : [{ img: j.img, map: j.map, groups: j.groups }];
      imagesCount = images_l.length;
      document.title = json.title;
      var p = new URLSearchParams(window.location.search);
      var si = parseInt(p.get("img")) || 0;
      if (si > 0 && si <= imagesCount) {
        current = si - 1;
        loadCurrentImage();
      } else if (imagesCount > 1) {
        loadIndex();
      } else {
        current = 0;
        loadCurrentImage();
      }
    })
    .fail(() => $("#h1").text("[*loading/error]1"));
});
