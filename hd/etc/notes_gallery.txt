<script>
function get( v ) {
  if( typeof v !== "undefined" ) {
    return v
  } else {
    return ""
  }
}
var doc_path="%prefix;m=DOC&s=";
var public_path="%bvar.gallery_path;";
var private_path="%bvar.gallery_path_private;";
$(document).ready(function() {
  $.ajax( "%url;ajax=on" )
  .done( function( json ) {
    $("#h1").text( json.title );
    if( json.path == "doc" ) {
      $("#image").attr( "src", doc_path+json.img );
    } else if( json.path == "private" ) {
      $("#image").attr( "src", private_path+json.img );
    } else if ( json.path == "public" ) {
      $("#image").attr( "src", public_path+json.img );
    } else {
      $("#image").attr( "src", json.path+json.img );
    }

    var cnt = 1;
    json.map.forEach( function( r ) {
      if( get(r.t) == "" || r.t == "p" ) {
        var href = "";
        var txt = "";
        var oc = get(r.oc);
        if( oc != "" && oc != "0" ) { oc = "&oc=" + oc }
        if( get(r.fn) != "" && get(r.sn) != "" ) {
          href = " href='%prefix;&p=" + r.fn + "&n=" + r.sn + oc + "'";
          txt = r.fn + " " + r.sn;
        }
        if( get(r.alt) != "" ) { txt += " (" + r.alt + ")"; }
        $("#map").append( "<area id='area" + cnt + "' shape='" + get(r.shape) + "' coords='" + get(r.coords) + "' alt='" + txt + "' title='" + txt + "'" + href + ">" );
        if( get(r.fn) != "" && get(r.sn) != "" ) {
          $("#legend").append( "<li class='legend' id='a" + cnt + "'><span><a href='%prefix;&p=" + r.fn + "&n=" + r.sn + oc + "'>" + txt + "</a></span></li>" );
        } else {
          $("#unknown").append( "<li class='legend' id='a" + cnt + "'><span>" + cnt + "</span></li>" );
          $("#div_unknown").show();
        }
      } else if( r.t == "m" ) {
        $("#legend").append( "<li class='legend' id='a" + cnt + "'><span><a href='%prefix;m=NOTES&f=" + get(r.misc) + "'>" + get(r.alt) + "</a></span></li>" );
      } else if( r.t == "g" ) {
        $("#legend").append( "<li class='legend' id='a" + cnt + "'><span><a href='%prefix;" + get(r.href) + "'>" + get(r.alt) + "</a></span></li>" );
      } else if( r.t == "e" ) {
        $("#legend").append( "<li class='legend' id='a" + cnt + "'><span><a href='" + get(r.href) + "'>" + get(r.alt) + "</a></span></li>" );
      }
      cnt++;
    });

    $("#image").maphilight();

    $(".legend").mouseover( function(e) {
      $( "#area" + $(this).attr("id").replace(/[a-z]/g, "") ).mouseover();
    }).mouseout( function(e) {
      $( "#area" + $(this).attr("id").replace(/[a-z]/g, "") ).mouseout();
    });

    $("area").mouseover( function(e) {
      $( "#a" + $(this).attr("id").replace(/[a-z]/g, "") + " span" ).addClass( "in" );
    }).mouseout( function(e) {
      $( "#a" + $(this).attr("id").replace(/[a-z]/g, "") + " span" ).removeClass( "in" );
    });
  })
  .fail( function() {
    $("#h1").text( "Erreur de chargement" );
  })
} );
</script>
