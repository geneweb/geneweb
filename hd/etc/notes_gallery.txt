<script>
var doc_path="%prefix;m=DOC&s=";
var public_path="%bvar.gallery_path;";
var private_path="%bvar.gallery_path_private;";
$(document).ready(function() {
  $.ajax( "%url;ajax=on" )
  .done( function( json ) {
    $("#h1").text( json.title );
    if( json.path == "doc" ) {
      $("#image").attr( "src", doc_path+json.img );
    } else if( json.path == "private" ) {
      $("#image").attr( "src", private_path+json.img );
    } else if ( json.path == "public" ) {
      $("#image").attr( "src", public_path+json.img );
    } else {
      $("#image").attr( "src", json.path+json.img );
    }

    var cnt = 1;
    var sep = "";
    var sep_u = "Non identifi√©s : ";
    json.map.forEach( function( i ) {
      var href = "";
      if( i.fn != "" ) {
        href = " href='%prefix;&p=" + i.fn + "&n=" + i.sn + "&oc=" + i.oc + "'";
      }
      $("#map").append( "<area id='area" + cnt + "' shape='" + i.shape + "' coords='" + i.coords + "' alt='" + i.alt + "' title='" + i.alt + "'" + href + ">" );
      if( i.fn != "" ) {
        $("#legend").append( sep + "<a class='legend' id='a" + cnt + "' href='%prefix;&p=" + i.fn + "&n=" + i.sn + "&oc=" + i.oc + "'>" + i.alt + "</a>" );
        sep = ", ";
      } else {
        $("#unknown").append( sep_u + "<span class='legend' id='a" + cnt + "'>" + cnt + "</span>" );
        sep_u = ", ";
      }
      cnt++;
    });

    $("#image").maphilight();

    $(".legend").mouseover( function(e) {
      $( "#area" + $(this).attr("id").replace(/[a-z]/g, "") ).mouseover();
    }).mouseout( function(e) {
      $( "#area" + $(this).attr("id").replace(/[a-z]/g, "") ).mouseout();
    });

    $("area").mouseover( function(e) {
      $( "#a" + $(this).attr("id").replace(/[a-z]/g, "") ).addClass( "in" );
    }).mouseout( function(e) {
      $( "#a" + $(this).attr("id").replace(/[a-z]/g, "") ).removeClass( "in" );
    });
  })
  .fail( function() {
    $("#h1").text( "Erreur de chargement" );
  })
} );
</script>
