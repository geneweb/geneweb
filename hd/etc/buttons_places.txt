<script>
// Synchronise l'attribut `value` et la propriété `checked` d'une case
function toggle(btnId, state) {
  const el = document.getElementById(btnId);
  if (!el) return;
  el.checked = state;
  el.value = state ? "on" : "";
}

// Ids des 5 cases d'événements (ordre préservé)
const eventIds = ["bi", "ba", "ma", "de", "bu"];

// Met à jour la case "all" selon l'état des 5 cases d'événements.
// CORRECTIF : l'original testait `bp` deux fois et ignorait `bu`
// (bug silencieux hérité — la case "all" ne se décochait jamais
// quand seule "bu" était décochée). Validé pour correction.
function test_all() {
  const allChecked = eventIds.every(id => document.getElementById(id).checked);
  document.getElementById("all").checked = allChecked;
}

// Câblage des événements (séparation contenu/comportement)
document.addEventListener("DOMContentLoaded", () => {
  // Chaque case d'événement met à jour la case "all"
  eventIds.forEach(id => {
    const el = document.getElementById(id);
    if (el) el.addEventListener("change", test_all);
  });

  // La case "all" bascule toutes les cases d'événements
  const allEl = document.getElementById("all");
  if (allEl) {
    allEl.addEventListener("change", () => {
      const state = allEl.checked;
      eventIds.forEach(id => toggle(id, state));
    });
  }
});
</script>
