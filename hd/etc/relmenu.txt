<script>
function enable(id) {
  const inputs = document.getElementById(id).getElementsByTagName('input');
  for (let i = 0; i < inputs.length; i++) {
    inputs[i].disabled = false;
  }
}

function disable(id) {
  const inputs = document.getElementById(id).getElementsByTagName('input');
  for (let i = 0; i < inputs.length; i++) {
    if (inputs[i].name !== 'select') {
      inputs[i].disabled = true;
    }
  }
}

function main() {
  const radioR = document.getElementById('radio_R');
  if (radioR && radioR.checked === true) {
    disable('rel_form_shortcut');
    enable('rel_form_search');
  } else {
    disable('rel_form_search');
    enable('rel_form_shortcut');
  }
}

document.addEventListener('DOMContentLoaded', main);
</script>
