<script>
(function () {
  const tbody = document.querySelector('#upd tbody') || document.querySelector('tbody');

  // ---------- helpers ----------
  const isVisible = (el) => el && el.offsetParent !== null;
  const visibleRows = () =>
    Array.from(tbody.querySelectorAll('tr')).filter(isVisible);
  const visibleNonLast = () =>
    visibleRows().filter(r => !r.classList.contains('last_row'));

  const rowIndex = (row) =>
    Array.prototype.indexOf.call(row.parentNode.children, row);

  const $ = (sel) => document.querySelector(sel);
  const $$ = (sel) => document.querySelectorAll(sel);

  // ---------- line numbers + arrows ----------
  function updateLineNumbers() {
    let n = 1;
    visibleNonLast().forEach(row => {
      const cell = row.querySelector('.line-number');
      if (cell) cell.textContent = n++;
    });
  }

  function updateArrowVisibility() {
    tbody.querySelectorAll('.row_move_up, .row_move_down')
         .forEach(b => b.style.visibility = 'visible');

    const rows = visibleRows();
    if (rows.length) {
      const upBtn = rows[0].querySelector('.row_move_up');
      if (upBtn) upBtn.style.visibility = 'hidden';
    }
    // jQuery `tbody tr:last` = dernier <tr> du tbody, indépendamment de :visible
    const all = tbody.querySelectorAll('tr');
    if (all.length) {
      const downBtn = all[all.length - 1].querySelector('.row_move_down');
      if (downBtn) downBtn.style.visibility = 'hidden';
    }
    updateLineNumbers();
  }

  function toggleIcons(buttonSelector) {
    const btn = typeof buttonSelector === 'string' ? $(buttonSelector) : buttonSelector;
    if (!btn) return;
    btn.querySelectorAll('.toggle-icon').forEach(i => i.classList.toggle('d-none'));
  }

  // ---------- template cloning ----------
  function cloneTemplate() {
    const tpl = document.querySelector('.template_line').cloneNode(true);
    tpl.classList.remove('template_line', 'd-none');
    return tpl;
  }

  // ---------- delegated click handler (row ops) ----------
  document.addEventListener('click', (e) => {
    const btn = e.target.closest('button, .row_add, .row_del, .row_move_up, .row_move_down');
    if (!btn) return;

    if (btn.classList.contains('row_add')) {
      const newRow = cloneTemplate();
      if (btn.closest('tfoot')) {
        tbody.appendChild(newRow);
      } else {
        const ref = btn.closest('tr');
        ref.parentNode.insertBefore(newRow, ref);
      }
      const firstInput = newRow.querySelector('input');
      if (firstInput) firstInput.focus();
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_del')) {
      btn.closest('tr').remove();
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_move_up')) {
      const row = btn.closest('tr');
      if (rowIndex(row) > 1) {
        const prev = row.previousElementSibling;
        if (prev) row.parentNode.insertBefore(row, prev);
        updateArrowVisibility();
      }
      return;
    }

    if (btn.classList.contains('row_move_down')) {
      const row = btn.closest('tr');
      const vis = visibleRows();
      const lastBody = vis[vis.length - 1];
      if (row !== lastBody) {
        const next = row.nextElementSibling;
        if (next) row.parentNode.insertBefore(row, next.nextElementSibling);
        updateArrowVisibility();
      }
    }
  });

  // ---------- delegated change handler ----------
  document.addEventListener('change', (e) => {
    const t = e.target;
    if (t.classList.contains('reset_bd_text')) {
      const cell = t.closest('tr').querySelector('.bd_text');
      if (cell) cell.innerHTML = '';
    }
  });

  // save RLM (localStorage)
  const saveBtn = $('#saveRLM');
  if (saveBtn) {
    saveBtn.addEventListener('click', () => {
      const src = $('#rlm-data');
      if (!src) return;
      const jsonText = src.innerHTML.trim();
      try {
        const jsonData = JSON.parse(jsonText);
        localStorage.setItem('inds', JSON.stringify(jsonData));
        toggleIcons('#saveRLM');
        setTimeout(() => toggleIcons('#saveRLM'), 2000);
      } catch (err) {
        console.error('Invalid JSON data in #rlm-data', err);
      }
    });
  }

  // ---------- URL builder ----------
  function buildUrl(mode) {
    mode = mode || 'RLM';
    let dataIndex = 1;
    const params = new URLSearchParams();
    params.set('m', mode);

    const currentParams = new URLSearchParams(window.location.search);
    const b = currentParams.get('b');
    if (b) params.set('b', b);

    visibleRows().forEach(row => {
      const pIn  = row.querySelector("input[name='p']");
      const nIn  = row.querySelector("input[name='n']");
      if (!pIn || !nIn) return;
      const firstName = pIn.value.trim();
      const lastName  = nIn.value.trim();
      if (firstName === '' && lastName === '') return;

      if (firstName !== '') params.set('p' + dataIndex, firstName);
      if (lastName  !== '') params.set('n' + dataIndex, lastName);

      const ocIn = row.querySelector("input[name='oc']");
      if (ocIn) {
        const oc = ocIn.value;
        if (oc !== '' && oc !== '0') params.set('oc' + dataIndex, oc);
      }
      if (mode === 'RLM') {
        const tIn = row.querySelector("input[name='t']");
        if (tIn) {
          const text = tIn.value.trim();
          if (text !== '') params.set('t' + dataIndex, text);
        }
      }
      dataIndex++;
    });

    return '?' + params.toString();
  }

  // ---------- top-level navigation buttons ----------
  const showGraph   = $('#showGraph');
  const showMatrix  = $('#showMatrix');

  if (showGraph) showGraph.addEventListener('click', () => {
    window.location.href = buildUrl('RLM');
  });
  if (showMatrix) showMatrix.addEventListener('click', () => {
    window.location.href = buildUrl('RM');
  });

  // form validity
  function checkFormValidity() {
    const warningRows = tbody.querySelectorAll('tr.row-invalid').length;
    const validRows = visibleRows().length - warningRows;
    const enable = warningRows === 0 && validRows >= 2;
    ['#saveRLM', '#showGraph', '#showMatrix'].forEach(sel => {
      const el = $(sel);
      if (el) el.disabled = !enable;
    });
  }

  // pnoc validation (live, on edit only)
  function rowKey(row) {
    const p  = row.querySelector("input[name='p']");
    const n  = row.querySelector("input[name='n']");
    const oc = row.querySelector("input[name='oc']");
    return {
      fn: p  ? p.value.trim()  : '',
      sn: n  ? n.value.trim()  : '',
      oc: oc ? (oc.value.trim() || '0') : '0'
    };
  }

function validateRow(row, opts) {
    if (!row || row.classList.contains('last_row')) return Promise.resolve();
    const k = rowKey(row);
    if (!k.fn && !k.sn) {
      row.classList.remove('row-invalid');
      delete row.dataset.pnocChecked;
      checkFormValidity();
      return Promise.resolve();
    }
    if (!k.fn || !k.sn) {
      row.classList.add('row-invalid');
      delete row.dataset.pnocChecked;
      checkFormValidity();
      return Promise.resolve();
    }
    const sig = k.fn + '|' + k.sn + '|' + k.oc;
    if (row.dataset.pnocChecked === sig) return Promise.resolve();
    return PersonPicker.checkExact(k.fn, k.sn, k.oc).then(ok => {
      row.dataset.pnocChecked = sig;
      row.classList.toggle('row-invalid', !ok);
      if (ok && opts && opts.flash) {
        row.classList.add('row-valid-flash');
        setTimeout(() => row.classList.remove('row-valid-flash'), 900);
      }
      checkFormValidity();
    });
  }

  document.addEventListener('focusout', (e) => {
    const inp = e.target.closest("tbody input[name='p'], tbody input[name='n'], tbody input[name='oc']");
    if (!inp) return;
    validateRow(inp.closest('tr'), { flash: true });
  });

  // active row tracking
  document.addEventListener('focusin', (e) => {
    const inp = e.target.closest(
      "tbody input[name='p'], tbody input[name='n'], " +
      "tbody input[name='oc'], tbody input[name='t']");
    if (!inp) return;
    const row = inp.closest('tr');
    if (!row) return;
    tbody.querySelectorAll('tr.row-active').forEach(r => {
      if (r !== row) r.classList.remove('row-active');
    });
    row.classList.add('row-active');
  });

  // init
function init() {
    updateArrowVisibility();
    checkFormValidity();
    setTimeout(() => {
      const firstWarning = tbody.querySelector(
        'tr.row-invalid input[name="p"]');
      if (firstWarning) {
        firstWarning.focus();
        if (firstWarning.setSelectionRange) {
          const len = firstWarning.value.length;
          firstWarning.setSelectionRange(len, len);
        }
      } else if (saveBtn) {
        saveBtn.focus();
      }
    }, 300);
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
</script>
</body>
