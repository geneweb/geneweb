<script>
(function () {
  const tbody = document.querySelector('#upd tbody');
  const form  = document.getElementById('upd');

  // ---------- helpers ----------
  const visibleRows = () =>
    Array.from(tbody.querySelectorAll('tr')).filter(r => r.offsetParent !== null);

  const nextUntil = (el, sel) => {
    const out = [];
    let n = el.nextElementSibling;
    while (n && !n.matches(sel)) { out.push(n); n = n.nextElementSibling; }
    return out;
  };
  const prevMatching = (el, sel) => {
    let n = el.previousElementSibling;
    while (n && !n.matches(sel)) n = n.previousElementSibling;
    return n;
  };
  const nextMatching = (el, sel) => {
    let n = el.nextElementSibling;
    while (n && !n.matches(sel)) n = n.nextElementSibling;
    return n;
  };
  const rowIndex = (row) =>
    Array.prototype.indexOf.call(row.parentNode.children, row);

  // ---------- line numbers + arrows ----------
  function updateLineNumbers() {
    let n = 1;
    visibleRows().forEach(row => {
      const cell = row.querySelector('.line-number');
      if (!cell) return;
      if (row.classList.contains('row_p'))      cell.textContent = n++;
      else if (row.classList.contains('row_s')) cell.textContent = '';
    });
  }

  function updateArrowVisibility() {
    tbody.querySelectorAll('.row_move_up, .row_move_down')
         .forEach(b => b.style.visibility = 'visible');

    const rows    = visibleRows();
    const persons = rows.filter(r => r.classList.contains('row_p'));

    persons.forEach((row, i) => {
      if (i === 0)
        row.querySelector('.row_move_up').style.visibility = 'hidden';
      if (i === persons.length - 1)
        row.querySelector('.row_move_down').style.visibility = 'hidden';
    });

    persons.forEach(personRow => {
      const group = nextUntil(personRow, '.row_p').filter(r => r.classList.contains('row_s'));
      if (group.length <= 1) {
        group.forEach(r => r.querySelectorAll('.row_move_up, .row_move_down')
                             .forEach(b => b.style.visibility = 'hidden'));
      } else {
        group[0].querySelector('.row_move_up').style.visibility = 'hidden';
        group[group.length - 1].querySelector('.row_move_down').style.visibility = 'hidden';
      }
    });

    updateLineNumbers();
  }

  // ---------- template cloning ----------
  function cloneTemplate(id) {
    const tpl = document.getElementById(id).cloneNode(true);
    tpl.removeAttribute('id');
    tpl.removeAttribute('style');
    return tpl;
  }
  function insertOrAppend(triggerBtn, ...newRows) {
    if (triggerBtn.closest('tfoot')) {
      newRows.forEach(r => tbody.appendChild(r));
    } else {
      const ref = triggerBtn.closest('tr');
      newRows.forEach(r => ref.parentNode.insertBefore(r, ref));
    }
  }

  // ---------- delegated click handler ----------
  document.addEventListener('click', (e) => {
    const btn = e.target.closest('button');
    if (!btn) return;

    if (btn.classList.contains('row_add_p_before')) {
      const p = cloneTemplate('template_line_p');
      const s = cloneTemplate('template_line_s');
      insertOrAppend(btn, p, s);
      const firstInput = p.querySelector('input');
      if (firstInput) firstInput.focus();
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_add_s_before')) {
      const s = cloneTemplate('template_line_s');
      insertOrAppend(btn, s);
      const firstInput = s.querySelector('input');
      if (firstInput) firstInput.focus();
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_del')) {
      const row = btn.closest('tr');
      if (row.classList.contains('row_p')) {
        nextUntil(row, '.row_p').forEach(r => r.remove());
        row.remove();
      } else if (row.classList.contains('row_s')) {
        const prev = row.previousElementSibling;
        const next = row.nextElementSibling;
        if ((prev && prev.classList.contains('row_s')) ||
            (next && next.classList.contains('row_s'))) {
          row.remove();
        }
      }
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_move_up')) {
      const row = btn.closest('tr');
      if (row.classList.contains('row_s')) {
        const prev = row.previousElementSibling;
        if (rowIndex(row) > 2 && prev && prev.classList.contains('row_s')) {
          row.parentNode.insertBefore(row, prev);
        }
      } else if (row.classList.contains('row_p')) {
        if (rowIndex(row) > 3) {
          const target = prevMatching(row, '.row_p');
          if (!target) return;
          const tail   = nextUntil(row, '.row_p');
          target.before(row);
          let anchor = row;
          tail.forEach(r => {
            anchor.parentNode.insertBefore(r, anchor.nextElementSibling);
            anchor = r;
          });
        }
      }
      updateArrowVisibility();
      return;
    }

    if (btn.classList.contains('row_move_down')) {
      const row = btn.closest('tr');
      if (row.classList.contains('row_s')) {
        const next = nextMatching(row, '.row_s');
        if (next) next.after(row);
      } else if (row.classList.contains('row_p')) {
        const tail   = nextUntil(row, '.row_p');
        const target = nextMatching(row, '.row_p');
        if (target) {
          const targetTail = nextUntil(target, '.row_p');
          const lastOfTarget = targetTail.length ? targetTail[targetTail.length - 1] : target;
          lastOfTarget.after(row);
          let anchor = row;
          tail.forEach(r => {
            anchor.after(r);
            anchor = r;
          });
        }
      }
      updateArrowVisibility();
    }
  });

  // ---------- delegated change handler ----------
  document.addEventListener('change', (e) => {
    const t = e.target;
    if (t.classList.contains('reset_bd_text')) {
      const cell = t.closest('tr').querySelector('.bd_text');
      if (cell) cell.innerHTML = '';
    }
    if (t.classList.contains('reset_n_p_oc')) {
      t.closest('tr').querySelectorAll('.n_p_oc').forEach(c => c.innerHTML = '');
    }
  });

  // ---------- submit: renommage indexé ----------
  let bypass = false;
  form.addEventListener('submit', (e) => {
    if (bypass) return;
    e.preventDefault();
    let gi = 1;

    visibleRows().forEach(row => {
      if (row.classList.contains('row_p')) {
        const p  = row.querySelector("input[name='p']");
        const n  = row.querySelector("input[name='n']");
        const oc = row.querySelector("input[name='oc']");
        const t  = row.querySelector("input[name='t']");
        if (p && n && p.value.trim() !== '' && n.value.trim() !== '') {
          if (oc) oc.name = 'oc' + gi;
          p.name = 'p' + gi;
          n.name = 'n' + gi;
          if (t) t.name = 't' + gi;
        } else {
          row.querySelectorAll('input').forEach(i => i.disabled = true);
        }
      } else if (row.classList.contains('row_s')) {
        const s = row.querySelector("input[name='s']");
        if (s && s.value.trim() !== '') {
          s.name = 's' + gi;
          gi++;
        } else {
          row.querySelectorAll('input').forEach(i => i.disabled = true);
        }
      }
    });

    bypass = true;
    form.submit();
  });

  // ---------- init ----------
  document.addEventListener('DOMContentLoaded', updateArrowVisibility);
  if (document.readyState !== 'loading') updateArrowVisibility();
})();
</script>
