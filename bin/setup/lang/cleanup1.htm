%h

[For information, this operation corresponds to the following commands][:] 

     $ cd "%w"
     $ %x%/gwu %a -o tmp.gw
     $ mkdir old
     [Under Unix][:]
        $ rm -rf old/%a.gwb
        $ mv %a.gwb old/.
     [Under MSdos][:]
        $ del old\%a.gwb\*.*
        $ rmdir old\%a.gwb
        $ move %a.gwb old\.
     $ %x%/gwc tmp.gw -nofail -o %a > comm.log
